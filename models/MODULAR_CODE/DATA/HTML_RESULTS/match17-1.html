#include<iostream>
using namespace std;
class Node
{
    private: 
            int data;
            Node* next;
    public: 
            Node(int data)
            {
                this->data = data;
                next = NULL;                
            }
            
            int getdata()
            {
                return data;
            }
            Node* getnext()
            {
                return next;
            }
            void setnext(Node* next)
            {
                this->next = next;
            }
};
