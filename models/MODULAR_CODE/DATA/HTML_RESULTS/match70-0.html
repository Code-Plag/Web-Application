#include<iostream>

using namespace std;

class Node
{
    private:
            int data;
            Node* next;
    public:
            Node()
            {
                next = NULL;
                //data = 0;
            }
        
            Node(int data)
            {
                this->data=data;
                this->next=NULL;
            }
    
            int getData()
            {
                return data;
            }

            Node* getNext()
            {
                return next;
            }
            void setNext(Node* next)
            {
                this->next=next;
            }

};
